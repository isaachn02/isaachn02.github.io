-- =====================================================
-- EXAMEN 1: CURASANO_DB

DROP DATABASE IF EXISTS CuraSano_db;
CREATE DATABASE CuraSano_db;
USE CuraSano_db;

-- =====================================================
-- EJERCICIO 01
-- Crear tablas
-- =====================================================

CREATE TABLE Medicos (
    idMedico INT AUTO_INCREMENT PRIMARY KEY,
    nombre VARCHAR(80) NOT NULL,
    especialidad VARCHAR(30) NOT NULL,
    turno CHAR(10) NOT NULL
);

CREATE TABLE Salas (
    idSala INT AUTO_INCREMENT PRIMARY KEY,
    codigo VARCHAR(20) NOT NULL UNIQUE,
    tipo VARCHAR(30) NOT NULL,
    capacidad INT NULL
);

CREATE TABLE Consultas (
    idConsulta INT AUTO_INCREMENT PRIMARY KEY,
    fechaHora DATETIME NOT NULL,
    duracion INT NOT NULL,
    estado ENUM('pendiente','finalizada','cancelada') NOT NULL,
    idMedico INT NOT NULL,
    idSala INT NOT NULL,
    FOREIGN KEY (idMedico) REFERENCES Medicos(idMedico),
    FOREIGN KEY (idSala) REFERENCES Salas(idSala)
);

-- Modificaciones

ALTER TABLE Consultas
ADD observaciones TEXT NULL;

ALTER TABLE Salas
ADD CONSTRAINT chk_capacidad
CHECK (capacidad <= 50);

ALTER TABLE Medicos
DROP COLUMN turno;

-- =====================================================
-- EJERCICIO 02
-- Consultas
-- =====================================================

-- 1. Mostrar identificador, fecha y código de sala,
-- ordenado por fecha descendente

SELECT c.idConsulta, c.fechaHora, s.codigo
FROM Consultas c
JOIN Salas s ON c.idSala = s.idSala
ORDER BY c.fechaHora DESC;

-- 2. Médicos con más de 5 consultas registradas

SELECT c.idMedico, COUNT(c.idConsulta) AS total_consultas
FROM Consultas c
GROUP BY c.idMedico
HAVING COUNT(c.idConsulta) > 5;

-- 3. Médicos sin consultas registradas usando subconsulta

SELECT nombre, especialidad
FROM Medicos
WHERE idMedico NOT IN (
    SELECT idMedico
    FROM Consultas
);

-- 4. Actualizar consultas pendientes anteriores a la fecha actual

UPDATE Consultas
SET estado = 'finalizada'
WHERE estado = 'pendiente'
AND fechaHora < NOW();

-- =====================================================
-- EJERCICIO 03
-- Procedimiento almacenado
-- =====================================================

DELIMITER $$

CREATE PROCEDURE ultimaConsultaMedico(
    IN p_idMedico INT,
    OUT p_fechaHora DATETIME
)
BEGIN
    SELECT MAX(fechaHora)
    INTO p_fechaHora
    FROM Consultas
    WHERE idMedico = p_idMedico;
END $$

DELIMITER ;

-- Llamada de prueba

CALL ultimaConsultaMedico(1, @ultimaConsulta);
SELECT @ultimaConsulta;

Examen 2: DinerosBank DB

-- =====================================================
-- EXAMEN 2: DINEROSBANK_DB
-- =====================================================

DROP DATABASE IF EXISTS dinerosBank_db;
CREATE DATABASE dinerosBank_db;
USE dinerosBank_db;

-- =====================================================
-- EJERCICIO 01
-- Crear tablas
-- =====================================================

CREATE TABLE Empleados (
    idEmpleado INT AUTO_INCREMENT PRIMARY KEY,
    nombre VARCHAR(80) NOT NULL,
    salario DECIMAL(10,2) NOT NULL
);

CREATE TABLE Prestamos (
    idPrestamo INT AUTO_INCREMENT PRIMARY KEY,
    importe DECIMAL(10,2) NOT NULL,
    estado ENUM('pendiente','aprobado','rechazado') NOT NULL,
    idEmpleado INT NOT NULL,
    FOREIGN KEY (idEmpleado) REFERENCES Empleados(idEmpleado)
);

CREATE TABLE Pago (
    idPago INT AUTO_INCREMENT PRIMARY KEY,
    fechaPago DATE NOT NULL,
    importe DECIMAL(10,2) NOT NULL,
    metodo ENUM('transferencia','efectivo','tarjeta') NOT NULL,
    idPrestamo INT NOT NULL,
    FOREIGN KEY (idPrestamo) REFERENCES Prestamos(idPrestamo)
);

-- Modificaciones

ALTER TABLE Empleados
ADD fechaContratacion DATE;

ALTER TABLE Empleados
MODIFY salario DECIMAL(10,2) NULL;

RENAME TABLE Pago TO Pagos;

-- =====================================================
-- EJERCICIO 02
-- Consultas
-- =====================================================

-- 1. Préstamos superiores a 50.000 y aprobados

SELECT *
FROM Prestamos
WHERE importe > 50000
AND estado = 'aprobado'
ORDER BY importe ASC;

-- 2. Empleados con importe total de préstamos gestionados,
-- incluyendo empleados sin préstamos

SELECT e.idEmpleado, e.nombre, SUM(p.importe) AS total_importe
FROM Empleados e
LEFT JOIN Prestamos p ON e.idEmpleado = p.idEmpleado
GROUP BY e.idEmpleado, e.nombre;

-- 3. Fecha del pago más reciente para cada préstamo

SELECT idPrestamo, MAX(fechaPago) AS fecha_pago_mas_reciente
FROM Pagos
GROUP BY idPrestamo;

-- 4. Eliminar pagos realizados en efectivo

DELETE FROM Pagos
WHERE metodo = 'efectivo';

-- =====================================================
-- EJERCICIO 03
-- Función almacenada 1
-- Total de préstamos aprobados de un empleado
-- =====================================================

DELIMITER $$

CREATE FUNCTION totalPrestamosAprobados(
    p_idEmpleado INT
)
RETURNS INT
DETERMINISTIC
BEGIN
    DECLARE total INT;

    SELECT COUNT(*)
    INTO total
    FROM Prestamos
    WHERE idEmpleado = p_idEmpleado
    AND estado = 'aprobado';

    RETURN total;
END $$

DELIMITER ;

-- Consulta de prueba

SELECT totalPrestamosAprobados(1);

-- =====================================================
-- FUNCIÓN ALTERNATIVA
-- Total de préstamos de un empleado para un estado determinado
-- =====================================================

DELIMITER $$

CREATE FUNCTION totalPrestamosPorEstado(
    p_idEmpleado INT,
    p_estado VARCHAR(20)
)
RETURNS INT
DETERMINISTIC
BEGIN
    DECLARE total INT;

    SELECT COUNT(*)
    INTO total
    FROM Prestamos
    WHERE idEmpleado = p_idEmpleado
    AND estado = p_estado;

    RETURN total;
END $$

DELIMITER ;

-- Consulta de prueba

SELECT totalPrestamosPorEstado(1, 'aprobado');

-- =====================================================
-- PROCEDIMIENTO ALMACENADO ALTERNATIVO
-- Total de préstamos por empleado y estado usando OUT
-- =====================================================

DELIMITER $$

CREATE PROCEDURE totalPrestamosEstado(
    IN p_idEmpleado INT,
    IN p_estado VARCHAR(20),
    OUT p_total INT
)
BEGIN
    SELECT COUNT(*)
    INTO p_total
    FROM Prestamos
    WHERE idEmpleado = p_idEmpleado
    AND estado = p_estado;
END $$

DELIMITER ;

-- Llamada de prueba

CALL totalPrestamosEstado(1, 'aprobado', @total);
SELECT @total;
